```html
            <li>
                <a href="https://mau.se" target="_blank">Malmö universitet</a>

            <input type="submit" value="Skicka">
        </form>
    </body>
</html>
```

style.css
{: .code-header}

```css
body {
    background-color: darksalmon;
}

img {
    width: 300px;
    border: 10px solid white;
    box-shadow: 0 0 10px #999;
    margin: 10px;
}

p {
    background-color: darkseagreen;
}

#big {
    font-size: 70px;
}

.right {
    text-align: right;
}
```
